\begin{frame}{Actions and change}
  \begin{itemize}
  \item The state of the world continuously changes through the execution\\ of actions
  \item States can be described by sets of properties whose values change over time
    \bigskip\pause
  \item \structure{Reasoning tasks}
    \begin{itemize}\normalsize
    \item \structure{Projection}  \ predict the state of the world after executing an action sequence
      \smallskip
    \item \structure{Planning}    \ identify ways to change the world to a pre-defined\\ goal state
      \smallskip
    \item \structure{Explanation} \ explain observed discrepancies
    \end{itemize}
  \end{itemize}
\end{frame}
