\begin{frame}{Actions and Change}
  \begin{itemize}
  \item The state of the world continuously changes through the execution of actions
  \item States can be described by sets of properties whose values change over time
  \item Tasks
\begin{itemize}
\item \alert{projection}: predicting the state of the world after the execution of an action sequence;
\item \alert{planning}: identifying ways to change the world to a pre-defined state;
\item \alert{explaination}: explaining the discrepancies that have been observed.
\end{itemize}
\end{frame}
