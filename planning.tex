\begin{frame}{Automated planning}

  \begin{itemize}
  \item \structure{Planning} \
    compute an action sequence transforming the state of the world from the initial state to the goal state
    \medskip
  \item \structure{Action description}\normalsize
    \begin{itemize}
    \item \structure{Fluents} \ properties that change over time
    \item \structure{Actions} \ operations leading to state changes
      \begin{itemize}\small
      \item action preconditions
      \item action effects
      \end{itemize}
    \item \structure{Laws}
      \begin{itemize}\small
      \item \structure{State constraints} \ domain knowledge
      \item \structure{Inertia}           \ handle properties that do not change
      \item \structure{Ramification}      \ handle indirect effects of actions
      \end{itemize}
    \end{itemize}

  \item \structure{Problem instance}\normalsize
    \begin{itemize}
    \item \structure{Objects}' \ definitions
    \item \structure{Start and goal state} \ a set of fluents
    \end{itemize}
  \end{itemize}
\end{frame}
